[1] MODULE TYPES & STANDARDS
----------------------------------------------------------------------------
CORE MODULES
- Built-in modules (NO INSTALL)
- Examples: http, fs, path, os, url
- Usage:
  const http = require('http');

LOCAL MODULES
- User-created reusable files
- Usage:
  const myModule = require('./file');

THIRD-PARTY MODULES
- Installed using npm
- Examples: express, dotenv
- Usage:
  npm install express

-------------------------------
COMMONJS (CJS) vs ES MODULES
-------------------------------
CJS (.js)
- require()
- module.exports
- synchronous

ESM (.mjs or "type":"module")
- import
- export
- asynchronous

Example:
CJS:
  const fs = require('fs');

ESM:
  import fs from 'fs';

[4] STREAMS (HIGH PERFORMANCE)
----------------------------------------------------------------------------
CONCEPT:
- Process data in chunks (NOT full file in memory)

TYPES:
Readable  -> Read data (fs.createReadStream)
Writable  -> Write data (fs.createWriteStream)
Pipe      -> Connect streams

Example:
const fs = require('fs');

let rs = fs.createReadStream('bigfile.txt');
let ws = fs.createWriteStream('copy.txt');

rs.pipe(ws);

OUTPUT:
[Chunk Received]
Streaming Complete ✔

[7] BASIC HTTP SERVER
----------------------------------------------------------------------------
const http = require('http');

const server = http.createServer((req, res) => {
    res.write("Hello World");
    res.end();
});

server.listen(3000, () => {
    console.log("Server running on port 3000");
});

RUN:
node app.js
nodemon app.js

----------------------------------------------------------------------------

[8] SERVER WITH MULTIPLE HTML PAGES
----------------------------------------------------------------------------
const http = require('http');
const fs = require('fs');

const server = http.createServer((req, res) => {

    if (req.url === "/" || req.url === "/home") {
        fs.createReadStream('./index.html').pipe(res);

    } else if (req.url === "/about") {
        fs.createReadStream('./about.html').pipe(res);

    } else {
        res.write("404 Page Not Found");
        res.end();
    }
});

server.listen(3000);

RUN:
nodemon server.js
